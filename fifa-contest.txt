    function loadState() {
      try {
        const raw = localStorage.getItem(STORAGE_KEY);
        if (!raw) return createDefaultState();
        const parsed = JSON.parse(raw);
        if (!parsed || !Array.isArray(parsed.participants) || !Array.isArray(parsed.matches)) return createDefaultState();
        return parsed;
      } catch {
        return createDefaultState();
      }
    }

      localStorage.setItem(STORAGE_KEY, JSON.stringify(state));

    function formatRs(n) {
      const hasDecimal = Math.abs(n % 1) > 0.001;
      return "₹" + n.toLocaleString("en-IN", {
        minimumFractionDigits: hasDecimal ? 2 : 0,
        maximumFractionDigits: 2
      });
    }

    function getDriveFileId(url) {
      if (!url) return "";
      const drivePathMatch = url.match(/\/d\/([A-Za-z0-9_-]+)/);
      const driveQueryMatch = url.match(/[?&]id=([A-Za-z0-9_-]+)/);
      return drivePathMatch ? drivePathMatch[1] : (driveQueryMatch ? driveQueryMatch[1] : "");
    }

    function getPhotoCandidates(name) {
      const rawUrl = participantPhotos[name];
      if (!rawUrl) return [];
      const fileId = getDriveFileId(rawUrl);
      if (!fileId) return [encodeURI(rawUrl)];
      return [
        `https://lh3.googleusercontent.com/d/${fileId}`,
        `https://lh3.googleusercontent.com/u/0/d/${fileId}`,
        `https://drive.google.com/uc?export=download&id=${fileId}`,
        `https://drive.google.com/uc?export=view&id=${fileId}`,
        `https://drive.google.com/thumbnail?id=${fileId}&sz=w1200`,
        rawUrl
      ];
    }

    function createParticipantLabel(name, sizeClass = "sm") {
      const wrap = document.createElement("span");
      wrap.className = "name-with-avatar";

      const photoCandidates = getPhotoCandidates(name);
      if (photoCandidates.length) {
        const img = document.createElement("img");
        img.className = `avatar ${sizeClass}`;
        img.src = photoCandidates[0];
        img.alt = name;
        img.referrerPolicy = "no-referrer";
        img.dataset.photoIndex = "0";
        img.addEventListener("error", () => {
          const nextIndex = Number(img.dataset.photoIndex || "0") + 1;
          if (nextIndex < photoCandidates.length) {
            img.dataset.photoIndex = String(nextIndex);
            img.src = photoCandidates[nextIndex];
            return;
          }
          if (img.parentElement) img.replaceWith(createFallbackAvatar(name, sizeClass));
        });
        wrap.appendChild(img);
      } else {
        wrap.appendChild(createFallbackAvatar(name, sizeClass));
      }

      text.textContent = name;
      wrap.appendChild(text);
      return wrap;
    }

    function addParticipant(name) {
      const clean = name.trim();
      if (!clean) return;
      if (state.participants.includes(clean)) {
        alert("Participant already exists.");
        return;
      }
      state.participants.push(clean);
      state.matches.forEach(m => {
        m.winnerPredictions[clean] = "";
        m.firstScorerPredictions[clean] = "";
      });
      saveState();
      render();
    }

    function removeParticipant(name) {
      state.participants = state.participants.filter(p => p !== name);
      state.matches.forEach(m => {
        delete m.winnerPredictions[name];
        delete m.firstScorerPredictions[name];
      });
      saveState();
      render();
    }

    function computePayouts() {
      const stakePerParticipant = TOTAL_BET_PER_MATCH * fixtures.length;
      const participantStats = Object.fromEntries(
        state.participants.map(name => [name, { name, correct: 0, won: 0, net: -stakePerParticipant }])
      );

      const matchStats = fixtures.map(f => {
        const match = getMatchState(f.id);
        const winnerPool = state.participants.length * WINNER_BET_PER_MATCH;
        const firstScorerPool = state.participants.length * FIRST_SCORER_BET_PER_MATCH;
        const totalPool = winnerPool + firstScorerPool;
        const winnerResult = match.winnerResult;
        const firstScorerResult = match.firstScorerResult;

        let winnerWinners = [];
        let firstScorerWinners = [];
        let winnerPayoutPerWinner = 0;
        let firstScorerPayoutPerWinner = 0;

        if (winnerResult) {
          winnerWinners = state.participants.filter(name => match.winnerPredictions[name] === winnerResult);
          winnerPayoutPerWinner = winnerWinners.length ? winnerPool / winnerWinners.length : 0;
          winnerWinners.forEach(name => {
            participantStats[name].correct += 1;
            participantStats[name].won += winnerPayoutPerWinner;
            participantStats[name].net += winnerPayoutPerWinner;
          });
        }

        if (firstScorerResult) {
          firstScorerWinners = state.participants.filter(name => match.firstScorerPredictions[name] === firstScorerResult);
          firstScorerPayoutPerWinner = firstScorerWinners.length ? firstScorerPool / firstScorerWinners.length : 0;
          firstScorerWinners.forEach(name => {
            participantStats[name].correct += 1;
            participantStats[name].won += firstScorerPayoutPerWinner;
            participantStats[name].net += firstScorerPayoutPerWinner;
          });
        }

        return {
          id: f.id,
          home: f.home,
          away: f.away,
          winnerResult: winnerResult || "Pending",
          winnerWinners,
          firstScorerResult: firstScorerResult || "Pending",
          firstScorerWinners,
          totalPool,
          winnerPayoutPerWinner,
          firstScorerPayoutPerWinner
        };
      });

      return {
        stakePerParticipant,
        matchStats,
        participantStats: Object.values(participantStats).sort((a, b) => b.won - a.won)
      };
    }

          chip.style.backgroundImage = `url('${fanFlagBg}')`;

        const btn = document.createElement("button");
        btn.textContent = "x";
        btn.addEventListener("click", () => {
          if (confirm(`Remove participant "${name}"?`)) removeParticipant(name);
        });
        chip.appendChild(btn);
        chips.appendChild(chip);
      });
    }

    function renderPayoutTables() {
      const { stakePerParticipant, matchStats, participantStats } = computePayouts();
      const done = matchStats.filter(m => m.winnerResult !== "Pending").length;
      document.getElementById("kpiParticipants").textContent = state.participants.length;
      document.getElementById("kpiPool").textContent = formatRs(state.participants.length * TOTAL_BET_PER_MATCH * fixtures.length);
      document.getElementById("kpiDone").textContent = `${done} / ${fixtures.length}`;

      const matchBody = document.querySelector("#matchPayoutTable tbody");
      matchBody.innerHTML = "";
      matchStats.forEach(m => {
        const tr = document.createElement("tr");
        const matchTd = document.createElement("td");
        matchTd.innerHTML = getFixtureLabelHtml(m.home, m.away);
        const winnerResultTd = document.createElement("td");
        winnerResultTd.innerHTML = m.winnerResult === "Pending" ? "Pending" : getTeamLabelHtml(m.winnerResult);
        const winnerWinnersTd = document.createElement("td");
        winnerWinnersTd.textContent = m.winnerResult === "Pending" ? "-" : (m.winnerWinners.length ? m.winnerWinners.join(", ") : "-");
        const firstScorerResultTd = document.createElement("td");
        firstScorerResultTd.innerHTML = m.firstScorerResult === "Pending"
          ? "Pending"
          : (m.firstScorerResult === "No goal" ? "No goal" : getTeamLabelHtml(m.firstScorerResult));
        const firstScorerWinnersTd = document.createElement("td");
        firstScorerWinnersTd.textContent = m.firstScorerResult === "Pending" ? "-" : (m.firstScorerWinners.length ? m.firstScorerWinners.join(", ") : "-");
        const totalPoolTd = document.createElement("td");
        totalPoolTd.textContent = formatRs(m.totalPool);
        const winnerPayoutTd = document.createElement("td");
        winnerPayoutTd.textContent = m.winnerResult === "Pending" ? "-" : formatRs(m.winnerPayoutPerWinner);
        const firstScorerPayoutTd = document.createElement("td");
        firstScorerPayoutTd.textContent = m.firstScorerResult === "Pending" ? "-" : formatRs(m.firstScorerPayoutPerWinner);
        tr.appendChild(matchTd);
        tr.appendChild(winnerResultTd);
        tr.appendChild(winnerWinnersTd);
        tr.appendChild(firstScorerResultTd);
        tr.appendChild(firstScorerWinnersTd);
        tr.appendChild(totalPoolTd);
        tr.appendChild(winnerPayoutTd);
        tr.appendChild(firstScorerPayoutTd);
        const winnerNoWinner = m.winnerResult !== "Pending" && m.winnerWinners.length === 0;
        const firstScorerNoWinner = m.firstScorerResult !== "Pending" && m.firstScorerWinners.length === 0;
        if (winnerNoWinner || firstScorerNoWinner) tr.className = "warn";
        else if (m.winnerResult !== "Pending" || m.firstScorerResult !== "Pending") tr.className = "ok";
        matchBody.appendChild(tr);
      });

      const contestantBody = document.querySelector("#contestantTable tbody");
      contestantBody.innerHTML = "";
      participantStats.forEach(p => {
        const tr = document.createElement("tr");
        const nameTd = document.createElement("td");
        nameTd.appendChild(createParticipantLabel(p.name, "sm"));
        const fanCountryTd = document.createElement("td");
        const fanCountry = getParticipantFanCountry(p.name);
        fanCountryTd.innerHTML = fanCountry ? getTeamLabelHtml(fanCountry) : "-";
        const correctTd = document.createElement("td");
        correctTd.textContent = p.correct;
        const wonTd = document.createElement("td");
        wonTd.textContent = formatRs(p.won);
        const netTd = document.createElement("td");
        netTd.textContent = formatRs(p.net);
        tr.appendChild(nameTd);
        tr.appendChild(fanCountryTd);
        tr.appendChild(correctTd);
        tr.appendChild(wonTd);
        tr.appendChild(netTd);
        contestantBody.appendChild(tr);
      });
    }

    function normalizeTeamValue(value) {
      if (value === "Winner Match 87" || value === "Columbia") return "Colombia";
      return value || "";
    }

    state.matches = fixtures.map(f => {
      const existing = state.matches.find(m => m.id === f.id);
      if (!existing) return {
        id: f.id,
        winnerResult: "",
        firstScorerResult: "",
        winnerPredictions: {},
        firstScorerPredictions: {}
      };
      const winnerPredictionsRaw = existing.winnerPredictions || existing.predictions || {};
      const firstScorerPredictionsRaw = existing.firstScorerPredictions || {};
      const winnerPredictions = Object.fromEntries(
        Object.entries(winnerPredictionsRaw).map(([name, pick]) => [name, normalizeTeamValue(pick)])
      );
      const firstScorerPredictions = Object.fromEntries(
        Object.entries(firstScorerPredictionsRaw).map(([name, pick]) => [name, normalizeTeamValue(pick)])
      );
      return {
        id: f.id,
        winnerResult: normalizeTeamValue(existing.winnerResult || existing.result || ""),
        firstScorerResult: normalizeTeamValue(existing.firstScorerResult || ""),
        winnerPredictions,
        firstScorerPredictions
      };
    });

    state.participants = Array.from(new Set(state.participants.map(p => (p || "").trim()).filter(Boolean)));
    if (!state.participants.includes("Jubil James")) {
      state.participants.push("Jubil James");
      state.matches.forEach(m => {
        m.winnerPredictions["Jubil James"] = m.winnerPredictions["Jubil James"] || "";
        m.firstScorerPredictions["Jubil James"] = m.firstScorerPredictions["Jubil James"] || "";
      });
    }
    if (!state.participants.includes("Fajer Yousufali")) {
      state.participants.push("Fajer Yousufali");
      state.matches.forEach(m => {
        m.winnerPredictions["Fajer Yousufali"] = m.winnerPredictions["Fajer Yousufali"] || "";
        m.firstScorerPredictions["Fajer Yousufali"] = m.firstScorerPredictions["Fajer Yousufali"] || "";
      });
    }
    saveState();
    render();
